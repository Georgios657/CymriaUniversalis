better_fertility = {
	icon = estate_god_privilege_giant_temples
	is_valid = {
		any_owned_province = {
		OR = {
			trade_goods = wool
			trade_goods = fish
			trade_goods = fur
			trade_goods = silk
			trade_goods = livestock
		}

		}		
	}
			max_absolutism = -5
			
		land_share = 0
		influence = 0.05
		loyalty = 0.0
		
			on_granted = {
		custom_tooltip = fertility_edict
		hidden_effect = {
			every_owned_province = {
				limit = {
		OR = {
			trade_goods = wool
			trade_goods = fish
			trade_goods = fur
			trade_goods = silk
			trade_goods = livestock
		}
				}
				add_province_modifier = {
					name = fertility_love
					duration = -1
				}
			}
		}			
	}

		on_granted_province = {
			if = {
				limit = {
		OR = {
			trade_goods = wool
			trade_goods = fish
			trade_goods = fur
			trade_goods = silk
			trade_goods = livestock
		}
				}
				add_province_modifier = {
					name = fertility_love
					duration = -1
				}
			}
		}	
		
	benefits = { 
	}
	
	
	penalties = {
		
	}
	
	on_revoked = { 
			custom_tooltip = fertility_edict_rev
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = fertility_love
		
					}
				}
				remove_province_modifier = fertility_love
			}
		}	
	
	add_estate_loyalty = {
    estate = estate_god_love
    loyalty = -20
	}

	}
	
	ai_will_do = {
		factor = 5
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}	
}
