the_holy_mountains = {
	icon = estate_god_telcontar_privilege_holy_swamp
	max_absolutism = -10
	land_share = 5	
	is_valid = {
		any_owned_province = {
		OR = {
			has_terrain = mountain
		}
		}
	
		
	}
	
	on_granted = {
		custom_tooltip = holy_mountains
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_terrain = mountain	
				}
				add_province_modifier = {
					name = holy_m
					duration = -1
				}
			}
		}
	}
		on_granted_province = {
			if = {
				limit = {
					has_terrain = mountain	
				}
				add_province_modifier = {
					name = holy_m
					duration = -1
				}
			}
		}		

	
		mechanics = {
		exempt_from_seize_land
	}
	
	influence = 0.05
	benefits = { 
		
	}
	penalties = {
	
	}
	
	on_revoked = { 
			custom_tooltip = holy_mountains_rev
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = holy_m
		
					}
				}
				remove_province_modifier = holy_m
			}
		}
		
	add_estate_loyalty = {
    estate = estate_god_earth
    loyalty = -20
	}

	}
	
	ai_will_do = {

		factor = 100
		modifier = {
			factor = 0
			OR = {
				NOT = { crown_land_share = 30 }
				estate_influence = {
					estate = estate_god_earth
					influence = 95
				}
			}
			NOT = { has_estate_privilege = the_holy_mountains }
		}
		modifier = {
			factor = 0.8
			OR = {
				NOT = { crown_land_share = 35 }
				AND = {
					is_absolutism_active = yes
					NOT = { crown_land_share = 55 }
				}
			}
			NOT = { has_estate_privilege = the_holy_mountains }
		}
		modifier = {
			factor = 0.8
			is_absolutism_active = yes
		}
		modifier = {
			factor = 1.25
			NOT = { adm = 2 }
		}
	}
	}
