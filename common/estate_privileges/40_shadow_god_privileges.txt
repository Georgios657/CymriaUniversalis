the_blessed_lands_of_darkness_and_ice = {
	icon = estate_god_privilege_giant_temples
		max_absolutism = -5
	is_valid = {
		any_owned_province = {
		AND = {
			 	culture_group = varjoren
			has_climate = arctic
		}
		}		
	}
		land_share = 5
	influence = 0.05
	loyalty = 0.0
	
	
			on_granted = {
		custom_tooltip = thorrens_gift
		hidden_effect = {
			every_owned_province = {
				limit = {
						AND = {
			 	culture_group = varjoren
			has_climate = arctic
		}
				}
				add_province_modifier = {
					name = blessed_artic
					duration = -1
				}
			}
		}
	}
	
			on_granted_province = {
			if = {
				limit = {
		AND = {
			 	culture_group = varjoren
			has_climate = arctic
		}
				}
				add_province_modifier = {
					name = blessed_artic
					duration = -1
				}
			}
		}	
	
	benefits = { 
		
	}
	
		mechanics = {
		exempt_from_seize_land
	}
	
	
	penalties = {
	 	
	}
	
	on_revoked = { 
				custom_tooltip = blessed_artic_removed
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = blessed_artic
		
					}
				}
				remove_province_modifier = blessed_artic
			}
		}
		
	add_estate_loyalty = {
    estate = estate_shadow_god
    loyalty = -20
	}

	}
	
	ai_will_do = {
		factor = 5
		modifier = {
			factor = 0
			is_absolutism_active = yes
		}
	}		
}
