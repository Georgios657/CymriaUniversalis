holy_swamp = {
	icon = estate_god_telcontar_privilege_holy_swamp
	max_absolutism = -10
	land_share = 10	
	is_valid = {
		religion = telcontarian
		any_owned_province = {
		OR = {
			has_terrain = marsh
		}
		}
	
		
	}
	
	on_granted = {
		custom_tooltip = blessed_swamps_grant
		hidden_effect = {
			every_owned_province = {
				limit = {
					AND = {
					has_terrain = marsh	
					region = lovia_region
				}
				}
				add_province_modifier = {
					name = blessed_swamps
					duration = -1
				}
			}
		}
	}
		on_granted_province = {
			if = {
				limit = {
					AND = {
					has_terrain = marsh	
					region = lovia_region
				}
				}
				add_province_modifier = {
					name = blessed_swamps
					duration = -1
				}
			}
		}		

	
		mechanics = {
		exempt_from_seize_land
	}
	
	influence = 0.05
	benefits = { 
		
	}
	penalties = {
	
	}
	
	on_revoked = { 
			custom_tooltip = blessed_swamps_removed
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = blessed_swamps
		
					}
				}
				remove_province_modifier = blessed_swamps
			}
		}
		
	add_estate_loyalty = {
    estate = estate_god_telcontar
    loyalty = -20
	}

	}
	
	ai_will_do = {

		factor = 100
		modifier = {
			factor = 0
			OR = {
				NOT = { crown_land_share = 30 }
				estate_influence = {
					estate = estate_god_telcontar
					influence = 95
				}
			}
			NOT = { has_estate_privilege = holy_swamp }
		}
		modifier = {
			factor = 0.8
			OR = {
				NOT = { crown_land_share = 35 }
				AND = {
					is_absolutism_active = yes
					NOT = { crown_land_share = 55 }
				}
			}
			NOT = { has_estate_privilege = holy_swamp }
		}
		modifier = {
			factor = 0.8
			is_absolutism_active = yes
		}
		modifier = {
			factor = 1.25
			NOT = { adm = 2 }
		}
	}
	}
