		hidden_effect = {
			every_owned_province = {

					region = gloomwood_region
				}
				add_province_modifier = {
					name = vengeful_woods_mod
					duration = -1
				}
			}
		}			
	}

		on_granted_province = {
		if = {
				limit = {
					region = gloomwood_region
				}

					name = vengeful_woods_mod
					duration = -1
				}		
		}

		}	
	
	benefits = { 
		
	}
	
	
	penalties = {
	}
	
	on_revoked = { 
	
				custom_tooltip = vengeful_woods_mod_rev
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						has_province_modifier = vengeful_woods_mod
		
					}
				}
				remove_province_modifier = vengeful_woods_mod
			}
		}	
	
	
	add_estate_loyalty = {
    estate = estate_gloom
    loyalty = -20
	}

	}
	
	ai_will_do = {
