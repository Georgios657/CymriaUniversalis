# Lollard Rebels
######################################################################
dragon_rebels = {

	color = { 235 115 115 }

	area = nation 				# Sticks to provinces of their religion
	government = any 			# Desires Theocracy
	defection = none					# Will defect to the "best" alternative.
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_province	
	will_relocate = yes	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no	
	
	artillery = 0.0
	infantry = 0.7
	cavalry = 0.3
	
	morale = 1.5
	
	# Possible handle actions
	handle_action_negotiate = yes	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0
			NOT = { has_province_modifier = stalking_dragon }
		}
		modifier = {
			factor = 1000
			has_province_modifier = stalking_dragon
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.001
			has_province_modifier = "stalking_dragon"
		}
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = { local_autonomy = 50 }
	}
	siege_won_effect = {
		add_local_autonomy = 25
		add_devastation = 50
	}
	
	# Country scope
	can_negotiate_trigger = {
	}
	
	# Country scope
	can_enforce_trigger = {
	}
	
	# Localisation for their demands
	demands_description = "dragon_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		add_years_of_income = -2
		every_owned_province = {
			remove_province_modifier = stalking_dragon
		}
	}
}
