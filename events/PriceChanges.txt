# Faceting
country_event = {
	id = prices_cym.39
	title = prices_cym.39.t
	desc = prices_cym.39.d
	picture = TRADEGOODS_eventPicture
	goto = diamond_city
	fire_only_once = yes
	
	major = yes

	trigger = {
		has_institution = renaissance
		prestige = 50
		any_owned_province = {
			renaissance = 100
			OR = {
				trade_goods = glass
				trade_goods = gems
			}
			development = 15
			is_state = yes
			is_city = yes
		}
	}
	mean_time_to_happen = {
		months = 1
	}
	

	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					renaissance = 100
					OR = {
						trade_goods = glass
						trade_goods = gems
					}
					development = 15
					is_state = yes
					is_city = yes
					province_without_special_goods_produced_trigger = yes
					NOT = { province_id = 112 }
				}
				save_event_target_as = diamond_city
				save_global_event_target_as = diamond_city_global
			}
		}
	}
	
	option = {
		name = prices_cym.39.a
		event_target:diamond_city = {
			if = {
				limit = {
					NOT = { trade_goods = gems }
				}
				change_trade_goods = gems
			}
			add_permanent_province_modifier = {
				name = diamond_district
				duration = -1
			}
		}
		change_price = {
			trade_goods = gems
			key = FACETING
			value = 0.25
			duration = -1
		}		
	}
}
