country_event = {
	id = mage_ruler_humans.1
	title = mage_ruler_humans.1.t
	desc = {
		desc = mage_ruler_humans.1.desc

	}

	picture = RELIGION_eventPicture

	mean_time_to_happen = {
	months = 120
}


	trigger = {
			NOT = {
	has_ruler_flag = immortal	
	}
		ruler_has_personality = wizard
		 	ruler_age = 30
			
					NOT = {
					OR = {
		has_ruler_flag = immortal	
		has_ruler_flag = blood
		government = magocracy
		ruler_is_vampire = yes
		ruler_is_dragon = yes
		ruler_is_scorpion = yes
		ruler_is_undead = yes
		ruler_is_elv = yes
		ruler_is_jhinn = yes					
					}

		}
	}

	immediate = {
	set_ruler_flag = voices
	}

	option = {
		name = mage_ruler_humans.1.a					
			 	add_adm_power = -25

		}
	}
