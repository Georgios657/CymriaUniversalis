		is_year = 1500
		tag = ZB2

	}

	
	immediate = {
		
		}
		
	option = {
		name = zephy.16.a	
	if = {
		limit = {
			republican_tradition = 90
		}
		set_country_flag = sucesslful_revolt
		custom_tooltip = may_found_zephyrian_empire
        }

		limit = {
			republican_tradition = 50
			NOT = {
			republican_tradition = 90
			}
		}

	}

		limit = {
			NOT = {
			republican_tradition = 50
			}

		release_all_possible_countries = yes
