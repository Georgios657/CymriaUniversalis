country_event = {
	id = dragon_election.1
	title = dragon_election.1.t
	desc = dragon_election.1.d
	picture = ELECTION_REPUBLICAN_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		 has_reform = dragon_protectorate_reform 
	}
	
	immediate = {

	}
	
	after = {

	}
	
	option = {
		name = dragon_election.1.a
		ai_chance = {
			factor = 100
		}
		trigger = {
			republican_tradition = 85
			
		}
				change_adm = 1
				change_dip = 1
				change_mil = 1
				add_scaled_republican_tradition = -10
			random_list = {
			10 = {
				add_adm_power = 50
			}
			10 = {
				add_dip_power = 50
			}
			10 = {
				add_mil_power = 50
			}
		}
		
				if = {
			limit = {
				has_government_attribute = reelection_depowers_estates
			}
			add_estate_influence_modifier = {
				estate = all
				influence = -10
				duration = 1460
				desc = REELCTION_DEPOWER_ESTATES
			}
		}
				if = {
			limit = { has_government_attribute = elections_influence_estates }
			add_estate_loyalty = {
				estate = all
				loyalty = 5
			}
		}
				if = {
			limit = { has_government_attribute = reelection_increases_absolutism }
			tooltip = {
				add_absolutism = 10
			}
			custom_tooltip = ADD_ABSOLUTISM_BASED_ON_ELECTION_TERM
			hidden_effect = {
				for_variable_amount = {
					variable = election_term
					effect = "add_absolutism = 10"
				}
			}
			hidden_effect = {
				change_variable = {
					which = election_term
					value = 1
				}
			}
		}
				if = {
			limit = { has_government_attribute = elections_increase_factions_influence }
			add_influence_to_random_faction = {
				influence = 20
			}
		}
	}
	option = {
		name = dragon_election.1.b
		ai_chance = {
			factor = 25		}
			add_army_tradition = 10
		add_faction_influence = {
    faction = male_dragons
    influence = 25
}
		define_ruler = {
		dip = 2
		adm = 2
		mil = 4
		}			
	}
	
	
	option = {
		name = dragon_election.1.c
		add_faction_influence = {
    faction = female_dragons
    influence = 25
}		
		ai_chance = {
			factor = 25
		}
		define_ruler = {
		dip = 3
		adm = 3
		mil = 2
		female = yes
		}
		add_republican_tradition = 10
	}
}
