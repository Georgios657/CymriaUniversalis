country_event = {
	id = varjorjen.2
	title = varjorjen.2.t
	desc = varjorjen.2.desc
	picture = RELIGION_eventPicture
	is_triggered_only = yes

	trigger = { }
	immediate = { }
		
	option = {
		name = varjorjen.2.a
		if = {
			limit = { tag = VA0 }
			exile_ruler_as = { name = yutu }
		    country_event = { 
				days = 3000
				random = 1500
				id = varjorjen.3
			}
        }
		if = {
			limit = { tag = VA2 }
			exile_ruler_as = { name = akna }
			country_event = { 
				days = 1500
				random = 750
				id = varjorjen.4
			}
        }		
		if = {
			limit = {
				NOT = {
					OR = {
						tag = VA0
						tag = VA2
					}
				}
			}
			kill_ruler = yes
        }		

		add_government_power = {
			mechanic_type = varjorjen_hunger_mechanic
			power_type = varjorjen_hunger
			value = 40
		}
			
		define_ruler = {
			dynasty = ROOT
			culture = ROOT
			religion = ROOT
			max_random_adm = 5
			max_random_dip = 5
			max_random_mil = 5
		}
		add_stability = -6
	}
}
